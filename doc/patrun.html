                descend(
                  nextkeymap,

                  _.extend({},valitermissing),

              }

          match:{},
          data:top.d,
          find:top.f
        })

    self.toString = function() {
      var tree = _.isBoolean( arguments[0] ) ? arguments[0] : !!arguments[1]

      function indent(o,d) {
        for(var i = 0; i < d; i++ ) {
          o.push(' ')

      var str = []

      function walk(n,o,d,vs){
        var vsc

        if( void 0 !== n.d ) {
          indent(o,d)
          o.push(dstr(n.d))

          str.push( vs.join(', ')+' -> '+dstr(n.d))
        }
        if( n.k ) {
          o.push('\n')
          indent(o,d)
          o.push(n.k+':')
        }
        if( n.v ) {
          d++
          for( var p in n.v ) {
            o.push('\n')
            indent(o,d)
            o.push( p+' ->')

            vsc = _.clone(vs)
            vsc.push(n.k+'='+p)

            walk(n.v[p],o,d+1,vsc)
          }

          if( n.s ) {
            o.push('\n')
            indent(o,d)
            o.push( '* ->')

            vsc = _.clone(vs)
            walk(n.s,o,d+1,vsc)
          }
        }
      }

      var o = []
      walk(top,o,0,[])
      return tree ? o.join('') : str.join('\n')
    }
